package Validator;

import Exception.DateException;

/**
 * Validator für Datums-Strings im Format dd.mm.yyyy.
 * Achtung: BaseValidator<T,E> benötigt zwei Argumente!
 */
public class DateValidator extends BaseValidator<String, DateException> {

    @Override
    public void validate(String obj) throws DateException {
        errors.clear();
        if (obj == null) {
            String msg = "Eingabe ist null.";
            errors.add(msg);
            throw new DateException(msg);
        }
        String dateStr = obj.trim();

        // Überprüfen, ob das Format dd.mm.yyyy eingehalten wird
        if (!dateStr.matches("^\\d{2}\\.\\d{2}\\.\\d{4}$")) {
            String msg = "Datum muss im Format dd.mm.yyyy sein.";
            errors.add(msg);
            throw new DateException(msg);
        }
    }
}
